SP_DEPLOY-WEBAPP() {
  GetVar -pattern "Environment" -name str_Env -default "staging"
  GetVar -pattern "Version" -name str_Version -default "latest"

  Message "Starting deployment of version ${str_Version} to ${str_Env}"
  Task "deploy_tool --env ${str_Env} --version ${str_Version}"
  Manual "Verify deployment status in dashboard before proceeding"
  Task "echo Deployment complete for ${str_Env}"
}
