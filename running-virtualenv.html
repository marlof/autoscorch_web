Running in a VirtualEnv

Virtual Env -1


sudo apt-get install python-virtualenv
sudo apt-get install python-pip
pip freeze > requirements.txt
mkdir ~/scorchpe
virtualenv ~/scorchvm1/scorch --no-site-packages
source ~/scorchvm1/scorch/bin/activate

mkdir ~/scorchvm1/scorch/demo
cd ~/scorchvm1/scorch/demo
wget http://www.autoscorch.com/downloads/install && chmod a+x install && ./install

deactivate
